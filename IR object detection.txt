#define IR_SENSOR_PIN 27  
void setup() {
  Serial.begin(9600);
  pinMode(IR_SENSOR_PIN, INPUT);
}
void loop() {
  int irValue = digitalRead(IR_SENSOR_PIN);
  if (irValue == LOW) {
    Serial.println("Object detected!");
  } else {
    Serial.println("No object detected.");
  }
  delay(500);
}
