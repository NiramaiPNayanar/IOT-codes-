int buttonState = 1;         // variable for reading the pushbutton status
void setup() {
  Serial.begin(115200);
  pinMode(15, OUTPUT);
  pinMode(32, INPUT);
  
}
void loop() {
    buttonState = digitalRead(32);
  if (buttonState == 0) {
    // turn LED on:
    digitalWrite(15, HIGH);
    Serial.println(buttonState);
  } else {
    // turn LED off:
    digitalWrite(15, LOW);
    Serial.println(buttonState);
  } 
delay(500);
}
